def bin_search(pole,x):
    min=0
    maxx=len(pole)-1
    mid=(min+max)//2
    while min < max:
        if x > pole[mid]:
            min=mid+1
            mid=(min+maxx)//2
        if x < pole[mid]:
            max=mid-1
            mid=(min+maxx)//2
        if x==pole[mid]:
            x==pole[mid]
            return(mid)
    return(-1)
